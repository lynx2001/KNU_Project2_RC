# Git


```bash
# git 초기화
git config --global user.name "내 깃허브 닉네임"
git config --global user.email "내 깃허브 이메일"

# 시작
git init

# 연결
git remote add origin https://github.com/lynx2001/KNU_Project2_RC.git

# 연결 확인
git remote -v


# Push
git add .
git commit -m "Message"
git push origin main

# Pull
git pull origin main

### Push
# 1. 'dev'이라는 이름의 새 브랜치를 만들고 거기로 이동
git checkout -b dev

# 2. 내 코드들 스테이징 (이미 했다면 생략 가능)
git add .

# 3. 커밋 (이미 했다면 생략 가능)
git commit -m "Message"

# 4. 깃허브로 푸시 (이때 브랜치가 자동 생성됨!)
git push origin dev


# 원격 저장소 정보 업데이트
git fetch origin

# dev 브랜치로 이동 및 연결
git checkout -b dev origin/dev

# 연결 확인
git branch

### Pull
git pull origin dev


### Merge (마지막)
# 1. 메인 브랜치로 이동
git checkout main

# 2. 원격의 최신 상태 가져오기
git pull origin main

# 3. dev 브랜치의 내용을 main에 합치기
git merge dev

# 4. 합쳐진 결과를 다시 깃허브에 반영
git push origin main



# !!!!! 강제로 pull 받아서 초기화
git fetch origin
git reset --hard origin/main  # 팀원의 로컬 변경사항이 날아갈 수 있으니 주의!
```


# Gazebo


```bash
# 윈도우 전용
sudo apt update && sudo apt install zsh -y

# 0. pixi 가상환경 진입
pixi shell

# 1. 빌드 찌꺼기 삭제 (필수!)
rm -rf build install log

# 2. 환경에 맞춰 새로 빌드
colcon build --symlink-install

# 3. 바뀐 pixi.toml 설정을 위해 다시 설치
pixi install

# 4. 터미널 환경 새로고침
source install/setup.zsh

# 실행
pixi run sim


----
# 초기화
rm -rf build/ install/ log/
```
